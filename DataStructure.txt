//file  function
#include<stdio.h>
void main()
{
   FILE *fp;
   char ch;
   clrscr();
   //writing data into file
   fp =fopen("test.txt","w");
   while((ch=getchar())!=EOF)
   {
     fputc(ch,fp);
     printf("%c",ch);
   }
   fclose(fp);
   //reading data into file
   fp =fopen("test.txt","r");
   while((ch=getchar())!=EOF)
   {
     fputc(ch,fp);
     printf("%c",ch);
   }
   fclose(fp);
}
//other ways
#include<stdio.h>
#include<stdlib.h>
void main()
{
  FILE *fp;
  char ch;
  clrscr();

  //write
  fp=fopen("test.txt","w");
  fprintf(fp,"hello");
  if(fp==NULL)
  {
    printf("file not found");
  }
  //exit(1);
  fclose(fp);

  //read
  fp=fopen("test.txt","r");
  fscanf(fp,"%s",&ch);
  printf("%s",&ch);
  fclose(fp);
  getch();
}
